194 void FbxConverter::ExtractFrameData(FbxImporter* importer)
195 {
196  // 1フレームの時間を取得
197  FbxTime::EMode time_mode = scene_->GetGlobalSettings().GetTimeMode();
198  period_.SetTime(0, 0, 0, 1, 0, time_mode);
199 
200  int animation_stack_count = importer->GetAnimStackCount();
201  std::cout << "\n★テイク数：" << animation_stack_count << std::endl;
202 
203  for (int i = 0; i < animation_stack_count; i++)
204  {
205  FbxTakeInfo* take_info = importer->GetTakeInfo(i);
206  if (!take_info) continue; // デフォルトテイクなら無視
207 
208  // アニメーションフレーム数を取得
209  FbxTime animation_start_time = take_info->mLocalTimeSpan.GetStart();
210  FbxTime animation_stop_time = take_info->mLocalTimeSpan.GetStop();
211  animation_start_frame_num_ = (int)(animation_start_time.Get() / period_.Get());
212  animation_stop_frame_num_ = (int)(animation_stop_time.Get() / period_.Get());
213  all_animation_frame_num_ = animation_stop_frame_num_ - animation_start_frame_num_ + 1;
214 
215  break;
216  }
217  if (all_animation_frame_num_ > 0)
218  {
219  std::cout << "⇒アニメーション情報：有" << std::endl;
220  std::cout << "⇒START_FRAME_NUM：" << animation_start_frame_num_ << std::endl;
221  std::cout << "⇒STOP_FRAME_NUM：" << animation_stop_frame_num_ << std::endl;
222  std::cout << "⇒ALL_FRAME_NUM：" << all_animation_frame_num_ << std::endl;
223  }
224  else
225  {
226  std::cout << "⇒アニメーション情報：無" << std::endl;
227  }

587 void FbxConverter::ExtractUVSetData(int mesh_index, FbxMesh* mesh)
588 {

590  FbxStringList uv_set_name_array;
591  mesh->GetUVSetNames(uv_set_name_array);
592  md_bin_data_.getpMesh(mesh_index)->setUVSetArraySize(uv_set_name_array.GetCount());
593 
594  for (int i = 0; i < uv_set_name_array.GetCount(); i++)
595  {
596  // UV配列(インデックスと同じ並び)
597  FbxArray<FbxVector2> uv_set_array;
598  mesh->GetPolygonVertexUVs(uv_set_name_array.GetStringAt(i), uv_set_array);
599 
600  // UV数をインデックス数分確保
601  md_bin_data_.getpMesh(mesh_index)->getpUVSet(i)->setUVArraySize(uv_set_array.Size());
602 
603  // UV取得(軸ごとにV値を変換)
604  for (int j = 0; j < md_bin_data_.getpMesh(mesh_index)->getpUVSet(i)->getUVArraySize(); j++)
605  {
606  *md_bin_data_.getpMesh(mesh_index)->getpUVSet(i)->getpU(j) = (float)uv_set_array[j][0];
