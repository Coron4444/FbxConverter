389 void ExportFile::CreateFileName(std::string* file_path)
390 {
391  // ファイル名を抽出(パス区切り"\"と"/"の両方に対応)
392  auto temp_index = file_path->find_last_of("/"); // 最後のパス区切りのインデックスを取得
393  if (temp_index != std::string::npos)
394  {
395  // ファイル名(最後のパス区切りから後ろ)を抽出
396  file_name_ = file_path->substr(temp_index + 1);
397  }
398  else
399  {
400  temp_index = file_path->find_last_of("\\");
401  if (temp_index != std::string::npos)
402  {
403  // ファイル名(最後のパス区切りから後ろ)を抽出
404  file_name_ = file_path->substr(temp_index + 1);
405  }
406  else
407  {
408  // ファイル名を抽出
409  file_name_ = *file_path;
410  }
411  }
412 
413  temp_index = file_name_.find_last_of(".");
414  if (temp_index != std::string::npos)
415  {
416  // ファイル名から拡張子を除く
417  file_name_ = file_name_.substr(0, temp_index);
418  }
419 }
